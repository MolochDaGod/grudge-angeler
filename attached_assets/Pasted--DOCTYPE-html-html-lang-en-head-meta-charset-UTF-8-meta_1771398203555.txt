    <script>
        // Configuration
        const APP_CONFIG = {
            characterPageUrl: 'https://warlord-crafting-suite.vercel.app/home',  // Home dashboard after login
            puterAppUrl: 'https://grudge-auth-73v97.puter.site',
            guestRedirectUrl: 'https://warlord-crafting-suite.vercel.app/home',  // Direct to home dashboard
        };

        // Local storage keys
        const STORAGE_KEYS = {
            accounts: 'grudge_guest_accounts',
            currentUser: 'grudge_current_user',
            userType: 'grudge_user_type',  // 'guest', 'puter', or 'wallet'
        };

        // Initialize on load
        window.addEventListener('DOMContentLoaded', checkExistingAuth);

        function checkExistingAuth() {
            const currentUser = localStorage.getItem(STORAGE_KEYS.currentUser);
            const userType = localStorage.getItem(STORAGE_KEYS.userType);
            
            if (currentUser && userType) {
                // User already logged in, redirect to character page
                console.log(`Existing ${userType} session found, redirecting...`);
                redirectToCharacterPage();
            }
        }

        function switchTab(tab) {
            const tabs = document.querySelectorAll('.tab');
            const loginForm = document.getElementById('login-form');
            const registerForm = document.getElementById('register-form');
            
            tabs.forEach(t => t.classList.remove('active'));
            event.target.classList.add('active');
            
            if (tab === 'login') {
                loginForm.classList.remove('hidden');
                registerForm.classList.add('hidden');
            } else {
                loginForm.classList.add('hidden');
                registerForm.classList.remove('hidden');
            }
        }
        
        function togglePassword() {
            const activeWrapper = event.target.closest('.input-password-wrapper');
            const activeInput = activeWrapper.querySelector('input');
            activeInput.type = activeInput.type === 'password' ? 'text' : 'password';
        }

        function redirectToCharacterPage() {
            window.location.href = APP_CONFIG.characterPageUrl;
        }
        
        function guestLogin() {
            // Create temporary guest session
            const guestId = 'guest_' + Date.now();
            const guestData = {
                id: guestId,
                username: 'Guest',
                type: 'temporary',
                createdAt: new Date().toISOString()
            };
            
            localStorage.setItem(STORAGE_KEYS.currentUser, JSON.stringify(guestData));
            localStorage.setItem(STORAGE_KEYS.userType, 'guest');
            
            showMessage('Entering as guest...');
            setTimeout(redirectToCharacterPage, 500);
        }
        
        async function walletConnect() {
            showMessage('Opening Web3 wallet connection...');
            
            // TODO: Integrate Web3Auth or similar wallet provider
            // For now, simulate wallet connection
            setTimeout(() => {
                const walletAddress = '0x' + Math.random().toString(16).substr(2, 40);
                const walletData = {
                    id: walletAddress,
                    type: 'wallet',
                    address: walletAddress,
                    connectedAt: new Date().toISOString()
                };
                
                localStorage.setItem(STORAGE_KEYS.currentUser, JSON.stringify(walletData));
                localStorage.setItem(STORAGE_KEYS.userType, 'wallet');
                
                showMessage('Wallet connected!');
                setTimeout(redirectToCharacterPage, 1000);
            }, 1000);
        }
        
        async function puterLogin() {
            showMessage('Connecting to Puter...');
            
            try {
                if (typeof puter !== 'undefined') {
                    // Use Puter SDK if available
                    if (!puter.auth.isSignedIn()) {
                        await puter.auth.signIn();
                    }
                    
                    const user = await puter.auth.getUser();
                    const puterData = {
                        id: user.uuid || user.username,
                        username: user.username,
                        email: user.email,
                        type: 'puter',
                        connectedAt: new Date().toISOString()
                    };
                    
                    localStorage.setItem(STORAGE_KEYS.currentUser, JSON.stringify(puterData));
                    localStorage.setItem(STORAGE_KEYS.userType, 'puter');
                    
                    showMessage('Puter connected!');
                    setTimeout(redirectToCharacterPage, 500);
                } else {
                    // Fallback: redirect to Puter app
                    window.location.href = APP_CONFIG.puterAppUrl;
                }
            } catch (error) {
                console.error('Puter login failed:', error);
                showMessage('Puter connection failed. Redirecting to Puter app...');
                setTimeout(() => {
                    window.location.href = APP_CONFIG.puterAppUrl;
                }, 1500);
            }
        }
        
        function login() {
            const username = document.getElementById('username').value.trim();
            const password = document.getElementById('password').value;
            
            if (!username || !password) {
                showMessage('Please enter username and password', 'error');
                return;
            }
            
            // Get stored accounts
            const accounts = JSON.parse(localStorage.getItem(STORAGE_KEYS.accounts) || '{}');
            const account = accounts[username];
            
            if (!account) {
                showMessage('Account not found. Please register first.', 'error');
                return;
            }
            
            // Simple password check (in production, use proper hashing)
            if (account.password !== password) {
                showMessage('Invalid password', 'error');
                return;
            }
            
            // Login successful
            const userData = {
                id: account.id,
                username: account.username,
                type: 'registered',
                createdAt: account.createdAt,
                loginAt: new Date().toISOString()
            };
            
            localStorage.setItem(STORAGE_KEYS.currentUser, JSON.stringify(userData));
            localStorage.setItem(STORAGE_KEYS.userType, 'guest');
            
            showMessage(`Welcome back, ${username}!`);
            setTimeout(redirectToCharacterPage, 1000);
        }
        
        function register() {
            const username = document.getElementById('reg-username').value.trim();
            const password = document.getElementById('reg-password').value;
            
            if (!username || !password) {
                showMessage('Please enter username and password', 'error');
                return;
            }
            
            if (username.length < 3) {
                showMessage('Username must be at least 3 characters', 'error');
                return;
            }
            
            if (password.length < 4) {
                showMessage('Password must be at least 4 characters', 'error');
                return;
            }
            
            // Get existing accounts
            const accounts = JSON.parse(localStorage.getItem(STORAGE_KEYS.accounts) || '{}');
            
            if (accounts[username]) {
                showMessage('Username already exists', 'error');
                return;
            }
            
            // Create new account
            const accountId = 'user_' + Date.now();
            accounts[username] = {
                id: accountId,
                username: username,
                password: password, // In production, hash this!
                createdAt: new Date().toISOString(),
                characters: []
            };
            
            localStorage.setItem(STORAGE_KEYS.accounts, JSON.stringify(accounts));
            
            // Auto-login after registration
            const userData = {
                id: accountId,
                username: username,
                type: 'registered',
                createdAt: accounts[username].createdAt
            };
            
            localStorage.setItem(STORAGE_KEYS.currentUser, JSON.stringify(userData));
            localStorage.setItem(STORAGE_KEYS.userType, 'guest');
            
            showMessage(`Account created! Welcome, ${username}!`);
            setTimeout(redirectToCharacterPage, 1000);
        }

        function showMessage(text, type = 'success') {
            // Create or update message element
            let messageEl = document.getElementById('auth-message');
            if (!messageEl) {
                messageEl = document.createElement('div');
                messageEl.id = 'auth-message';
                messageEl.style.cssText = `
                    position: fixed;
                    top: 20px;
                    left: 50%;
                    transform: translateX(-50%);
                    padding: 12px 24px;
                    border-radius: 8px;
                    font-weight: 600;
                    z-index: 9999;
                    animation: slideDown 0.3s ease;
                `;
                document.body.appendChild(messageEl);
            }
            
            messageEl.textContent = text;
            messageEl.style.background = type === 'error' 
                ? 'linear-gradient(to right, #ef4444, #dc2626)' 
                : 'linear-gradient(to right, #059669, #16a34a)';
            messageEl.style.color = 'white';
            messageEl.style.boxShadow = type === 'error'
                ? '0 4px 12px rgba(239, 68, 68, 0.4)'
                : '0 4px 12px rgba(5, 150, 105, 0.4)';
            
            // Auto-hide after 3 seconds
            setTimeout(() => {
                if (messageEl) {
                    messageEl.style.animation = 'slideUp 0.3s ease';
                    setTimeout(() => messageEl.remove(), 300);
                }
            }, 3000);
        }
    </script>
